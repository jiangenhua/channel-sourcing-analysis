                  三组关键发现:
                  1. res_1080p ↔ dur_60s: 正相关 (高清的视频往往也长), 可作为 "高质量" 联合判据.
                  2. video_cnt ↔ avg_play: 负相关 (视频越多平均播放越低), 印证"少而精"原则.

                  阶段 1 · 类目级过滤
                  Time: O(N) · Cost: 低
                  • ✓ 保留 9 个高质量类目
                  • ✓ 整类剔除 Gaming/Film/AIGC
                  • ✓ People&Blogs 二次打标后再筛
                  • ✓ Shorts/Other 直接过滤

                  2
                  阶段 2 · 视频级过滤
                  Time: O(N) · Cost: 中
                  • ✓ 时长 ≥ 30s
                  • ✓ 分辨率 ≥ 720p (排除<1080p 时降权)
                  • ✓ 帧率 ∈ {24, 30, 60}
                  • ✓ play_num ≥ 1K (排除冷启动)
                  • ✓ is_aigc = 0
                  视频留存: 2.46%
                  3
                  阶段 3 · channel 级打分
                  Time: O(N log N) · Cost: 高
                  • ✓ 八维评分 (QualityScore)
                  • ✓ 异常账号识别 (5 类规则)
                  • ✓ Pareto 摘头部 Top 5%
                  • ✓ S 级直接入选, A 级抽样审核
                  • ✓ 输出 ≈ 8,424 channel
                  最终留存: 0.0004%

                  📋 完整 pipeline SQL 模板

                  -- 阶段 1: 视频级过滤 + 派生字段
WITH filtered_videos AS (
  SELECT
    author,
    unified_category,
    play_num,
    like_num,
    comment_num,
    channel_follower_count AS follower,
    width,
    height,
    duration,
    bit_rate,
    frame_rate,
    requirement_info['is_aigc'] AS is_aigc
  FROM data_sec_management_app.dwd_hy_video_raw_acquire_data_hi
  WHERE requirement_id IN (5148, 5152)
    AND tdbank_imp_date >= '2026010800'
    AND duration >= 30
    AND frame_rate IN (24, 30, 60)
    AND width * height >= 1280 * 720
    AND play_num >= 1000
    AND COALESCE(requirement_info['is_aigc'], 0) = 0
    AND unified_category IN (
      'Travel & Events / 旅行', 'Education / 教育',
      'Sports / 体育', 'Autos & Vehicles / 汽车',
      'Pets & Animals / 宠物与动物', 'Howto & Style / 生活时尚',
      'Science & Technology / 科技', 'Food / 美食',
      'Family / 家庭'
    )
),
-- 阶段 2: channel 维度聚合
channel_stats AS (
  SELECT
    author,
    unified_category,
    COUNT(*) AS video_cnt,
    SUM(play_num) AS total_play,
    MEDIAN(play_num) AS median_play,
    AVG(CAST(width >= 1920 AS INT)) * 100  AS res_1080p_pct,
    AVG(CAST(duration >= 60 AS INT)) * 100   AS dur_60s_pct,
    AVG(CAST(duration >= 180 AS INT)) * 100  AS dur_180s_pct,
    (SUM(like_num)+SUM(comment_num))*100.0
      / NULLIF(SUM(play_num), 0) AS engagement_pct,
    SUM(like_num)*100.0
      / NULLIF(SUM(play_num), 0) AS like_per_100play,
    AVG(play_num) / NULLIF(MAX(follower), 0) AS avg_play_per_follower,
    MAX(follower) AS follower
  FROM filtered_videos
  GROUP BY author, unified_category
),
-- 阶段 3: 八维质量评分
scored AS (
  SELECT *,
    0.20 * CASE
      WHEN res_1080p_pct >= 80 THEN 100
      WHEN res_1080p_pct >= 60 THEN 75
      WHEN res_1080p_pct >= 40 THEN 50
      ELSE 25
    END
    + 0.15 * CASE
      WHEN dur_60s_pct >= 70 THEN 100
      WHEN dur_60s_pct >= 50 THEN 75
      WHEN dur_60s_pct >= 30 THEN 50
      ELSE 25
    END
    + 0.15 * CASE
      WHEN engagement_pct >= 3 THEN 100
      WHEN engagement_pct >= 2 THEN 75
      WHEN engagement_pct >= 1 THEN 50
      ELSE 25
    END
    -- ... 其余 5 个维度类似 (略, 详见 REPORT.md)
    AS quality_score
  FROM channel_stats
)
SELECT *
FROM scored
WHERE quality_score >= 60     -- S/A 级
  AND video_cnt >= 50             -- 持续创作
  AND follower >= 10000          -- 行业认可
ORDER BY quality_score DESC, total_play DESC;
